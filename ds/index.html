Quick Reference

<!-- 1. Include the stylesheet -->
<link rel="stylesheet" href="styles.css">

<!-- 2. Use ds- prefixed classes -->
<button class="ds-btn ds-btn-md ds-btn-primary">Save</button>
<button class="ds-btn ds-btn-md ds-btn-danger ds-btn-borderless">Delete</button>

<input class="ds-input ds-input-md" placeholder="Email">
<textarea class="ds-textarea" placeholder="Message..."></textarea>

<!-- Checkbox -->
<label class="ds-checkbox">
  <input type="checkbox" class="ds-checkbox-input">
  <span class="ds-checkbox-indicator">
    <svg width="12" height="12" viewBox="0 0 12 12" fill="none">
      <path d="M2.5 6l2.5 2.5 4.5-5" stroke="white" stroke-width="1.5"
        stroke-linecap="round" stroke-linejoin="round"/>
    </svg>
  </span>
  <span class="ds-checkbox-label">Remember me</span>
</label>

<!-- Radio -->
<label class="ds-radio">
  <input type="radio" name="group" class="ds-radio-input">
  <span class="ds-radio-indicator"></span>
  <span class="ds-radio-label">Option A</span>
</label>

<!-- Switch -->
<button class="ds-switch-track ds-switch-lg ds-switch-off"
        role="switch" aria-checked="false">
  <span class="ds-switch-handle"></span>
</button>

<!-- Tabs -->
<div class="ds-tabs" role="tablist">
  <button class="ds-tab" role="tab" aria-selected="true">Tab 1</button>
  <button class="ds-tab" role="tab" aria-selected="false">Tab 2</button>
</div>

<!-- Segmented Control -->
<div class="ds-segmented" role="radiogroup" aria-label="View">
  <button class="ds-segmented-item" role="radio" aria-checked="true">Label</button>
  <button class="ds-segmented-item" role="radio" aria-checked="false">Label</button>
  <button class="ds-segmented-item" role="radio" aria-checked="false">Label</button>
</div>

<!-- Select (functional dropdown) -->
<div class="ds-select-wrapper">
  <button class="ds-select" aria-haspopup="listbox">
    <span>Label</span>
    <svg class="ds-select-chevron" width="16" height="16"
      viewBox="0 0 16 16" fill="none">
      <path d="M4 6l4 4 4-4" stroke="currentColor" stroke-width="1.5"
        stroke-linecap="round" stroke-linejoin="round"/>
    </svg>
  </button>
  <div class="ds-select-menu" role="listbox">
    <button class="ds-select-item" role="option">
      <span class="ds-select-item-label">Item</span>
    </button>
  </div>
</div>

<!-- Data Table -->
<div class="ds-table">
  <div class="ds-table-action-bar">
    <input class="ds-input ds-input-md" placeholder="Search" style="width:160px">
    <div class="ds-table-action-bar-spacer"></div>
    <button class="ds-btn ds-btn-md ds-btn-primary">Create</button>
  </div>
  <table>
    <thead>
      <tr>
        <th class="ds-table-col-checkbox"><!-- checkbox --></th>
        <th class="ds-table-col-icon"></th>
        <th>Name</th>
        <th class="ds-table-sortable ds-table-sorted">Date ↓</th>
        <th class="ds-table-col-action"></th>
      </tr>
    </thead>
    <tbody>
      <tr>
        <td class="ds-table-col-checkbox"><!-- checkbox --></td>
        <td class="ds-table-col-icon"><!-- icon --></td>
        <td class="ds-table-cell-highlight">Document 1</td>
        <td>10/09/21</td>
        <td class="ds-table-col-action"><!-- action btn --></td>
      </tr>
    </tbody>
  </table>
  <div class="ds-table-pagination">
    <button class="ds-table-page">&lt;</button>
    <button class="ds-table-page" aria-current="page">1</button>
    <button class="ds-table-page">2</button>
    <button class="ds-table-page">&gt;</button>
  </div>
</div>

<span class="ds-table-status ds-table-status-green">Confirmed</span>
